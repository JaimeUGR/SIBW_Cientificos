CREATE TABLE CIENTIFICOS(
  ID INT AUTO_INCREMENT PRIMARY KEY,
  NOMBRE VARCHAR(50) NOT NULL,
  APELLIDOS VARCHAR(150),
  FECHA_NACIMIENTO DATE,
  FECHA_MUERTE DATE,
  LUGAR_NACIMIENTO VARCHAR(50),
  PAIS_ORIGEN VARCHAR(50),
  DATA_PATH VARCHAR(250) NOT NULL
);

CREATE TABLE REDES_CIENTIFICO(
  NUM INT,
  ID INT,
  NOMBRE VARCHAR(50) NOT NULL,
  ENLACE VARCHAR(500) NOT NULL,
  PRIMARY KEY (ID, NUM),
  FOREIGN KEY (ID) REFERENCES CIENTIFICOS(ID)
);

CREATE TABLE BAD_WORDS(
  PALABRA VARCHAR(30) PRIMARY KEY
);

CREATE TABLE USUARIOS(
  USUARIO VARCHAR(30) PRIMARY KEY,
  PASSWORD VARCHAR(250) NOT NULL,
  EMAIL VARCHAR(200) UNIQUE NOT NULL,
  NOMBRE VARCHAR(50) NOT NULL,
  APELLIDOS VARCHAR(150),
  PERMISOS VARCHAR(30),
  FOREIGN KEY (PERMISOS) REFERENCES PERMISOS(PERMISO)
);

CREATE TABLE PERMISOS(
  PERMISO VARCHAR(30) PRIMARY KEY
);

CREATE TABLE COMENTARIOS(
  ID_CIENTIFICO INT,
  USUARIO VARCHAR(30),
  COMENTARIO MEDIUMTEXT NOT NULL,
  FECHA_COMENTARIO DATETIME,
  EDITADO BOOLEAN NOT NULL DEFAULT FALSE,
  PRIMARY KEY (ID_CIENTIFICO, USUARIO),
  FOREIGN KEY (ID_CIENTIFICO) REFERENCES CIENTIFICOS(ID),
  FOREIGN KEY (USUARIO) REFERENCES USUARIOS(USUARIO)
);

CREATE TABLE HASHTAGS_CIENTIFICO(
  HASHTAG VARCHAR(30),
  ID_CIENTIFICO INT,
  PRIMARY KEY (HASHTAG, ID_CIENTIFICO),
  FOREIGN KEY (ID_CIENTIFICO) REFERENCES CIENTIFICOS(ID)
);
